cmake_minimum_required(VERSION 3.10)
project(FiguresProject)

set(CMAKE_TOOLCHAIN_FILE "C:/vcpkg/scripts/buildsystems/vcpkg.cmake")

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include_directories(include)

set(SOURCES
    src/figure.cpp
    src/square.cpp
    src/rectangle.cpp
    src/trapezoid.cpp
)

set(MAIN_SOURCES
    main.cpp
    ${SOURCES}
)

add_executable(figures_program ${MAIN_SOURCES})


enable_testing()
find_package(GTest CONFIG REQUIRED) 

set(TEST_SOURCES
    test/test_figures.cpp
    ${SOURCES}
)

add_executable(figures_test ${TEST_SOURCES})
target_link_libraries(figures_test PRIVATE GTest::gtest GTest::gtest_main)

include(GoogleTest)
gtest_discover_tests(figures_test)
