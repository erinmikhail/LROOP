cmake_minimum_required(VERSION 3.10)
project(SixProject)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include_directories(include)

set(SOURCES
    src/six.cpp
)

set(MAIN_SOURCES
    main.cpp
    ${SOURCES}
)

add_executable(six_program ${MAIN_SOURCES})

# Google Test
enable_testing()
find_package(GTest REQUIRED)

set(TEST_SOURCES
    test/test_six.cpp
    ${SOURCES}
)

add_executable(six_test ${TEST_SOURCES})
target_link_libraries(six_test GTest::GTest GTest::Main)

include(GoogleTest)
gtest_discover_tests(six_test)
